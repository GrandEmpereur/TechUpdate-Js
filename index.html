To do this nav you need to do :


        nav{
            display: flex; 
            justify-content: space-around; /* set the div and ul with the same space */
            align-items: center; /* center the elements inside the parent */
            min-height: 8vh;/* minimum height possible set by hand */
            background-color: chocolate;
            color: whitesmoke;/* change the text color */
        }
        
        .title{
            text-transform: uppercase;/* switch lowercase to uppercase */
            letter-spacing: 2px;/* put a space between the letters */
            font-size: 20px;/* change the size of the font */
        }
        
        .nav-links{
            display: flex;
            width: 40%;/* change the width of the div */
            justify-content: space-around;
            letter-spacing: 1px;
            font-weight: bolder;/* change the state of the font */
        }
        
        .nav-links li{
            list-style: none;/* remove the dots in front of the list items */
        }
        
        .nav-links a{
            color: whitesmoke;
            font-size: 15px;
        }
        
        .burger-menu{
            display: none;/* make the burger menu disappear in desktop device */
            cursor: pointer;
        }

        .burger-menu div{
            width: 30px;/* length of the burger menu lines */
            height: 3px;/* height of the burger menu lines */
            background-color: whitesmoke;/* color of the burger menu lines */
            margin: 5px;/* distribute equitably within the div */
        }

To do the burger menu you need to do this :


        .nav-links{
            /* the targeted element becomes independent */
            position: absolute;
            /* put the div on the right */
            right: 0;
            /* minimum height possible set by hand for responsive */
            height: 92vh;
            /* put the div on the top */
            top: 8vh;
            background-color: chocolate;
            /* allows to add shadows on divs and the inset property allows to put the shadow inside the div */
            box-shadow: inset 0 5px 15px rgba(0, 0, 0, 0.329);
            display: flex;
            /* put the elements of the div in column */
            flex-direction: column;
            /* it defines the alignment of an object in its parent */
            align-items: center;
            width: 25%;
            /* move the elements according to the X axis */
            transform: translateX(100%);
            transition: 0.5s ease-in;
        }
        .nav-links li{
            opacity: 0;
        }
        .burger-menu{
            /* make the elements reappear */
            display: block;
        }

and for the javascript you need to do :


        const navSlide = () =>{
            const burger = document.querySelector('.burger-menu');
            const nav = document.querySelector('.nav-links');
            const navLinks = document.querySelectorAll('.nav-links li');
        
            burger.addEventListener('click',() => {
                nav.classList.toggle('nav-on');
        
            });
        
            navLinks.forEach((link,index) => {
                link.style.animation = `navLinksFade 0.5s ease forwards ${index / 7}s`;
                console.log(index / 7);
            });
        }
        
        navSlide();
